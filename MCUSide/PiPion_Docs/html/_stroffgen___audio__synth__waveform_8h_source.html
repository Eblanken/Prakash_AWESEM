53 class AudioSynthWaveform : public AudioStream

55 public:
56  AudioSynthWaveform(void) : AudioStream(0,NULL),
57  phase_accumulator(0), phase_increment(0), phase_offset(0),
58  magnitude(0), pulse_width(0x40000000), sample(0), tone_type(WAVEFORM_SINE),
59  tone_offset(0) {

62  void frequency(float freq) {
63  if (freq < 0.0) {
64  freq = 0.0;
65  } else if (freq > AUDIO_SAMPLE_RATE_EXACT / 2) {
66  freq = AUDIO_SAMPLE_RATE_EXACT / 2;
67  }
68  phase_increment = freq * (4294967296.0 / AUDIO_SAMPLE_RATE_EXACT);
69  if (phase_increment > 0x7FFE0000u) phase_increment = 0x7FFE0000;
70  }
71 
72  void phase(float angle) {
73  if (angle < 0.0) {
74  angle = 0.0;
75  } else if (angle > 360.0) {
76  angle = angle - 360.0;
77  if (angle >= 360.0) return;
78  }
79  phase_offset = angle * (4294967296.0 / 360.0);
80  }

82  /*
83  * Description:
84  * MODDED: Erick Blankenberg, sets so that there are a set number of blocks until overflow
85  */
86  void restart() {
87  const uint32_t numBlocksBeforeRollover = 6;
88  phase_accumulator = UINT32_MAX - (numBlocksBeforeRollover * AUDIO_BLOCK_SAMPLES * phase_increment);
89  }
90 
91  /*
92  * Description:
93  * MODDED: Erick Blankenberg, trying to starve dac output by cutting off source
94  * so that it reaches known initial ready state once it has exhausted all internal
95  * resources. This function prevents the waveform from progresssing and stops output.
96  */
97  void disableOutput() {
98  synthEnabled = false;
99  }

101  /*
102  * Description:
103  * MODDED: Erick Blankenberg, see above, this function enables progression of

108  }

110  void amplitude(float n) { // 0 to 1.0
111  if (n < 0) {
112  n = 0;
113  } else if (n > 1.0) {
114  n = 1.0;
115  }
116  magnitude = n * 65536.0;
117  }

119  void offset(float n) {
120  if (n < -1.0) {
121  n = -1.0;
122  } else if (n > 1.0) {
123  n = 1.0;
124  }
125  tone_offset = n * 32767.0;
126  }
127 
128  void pulseWidth(float n) { // 0.0 to 1.0
129  if (n < 0) {
130  n = 0;
131  } else if (n > 1.0) {
132  n = 1.0;
133  }
134  pulse_width = n * 4294967296.0;
135  }
